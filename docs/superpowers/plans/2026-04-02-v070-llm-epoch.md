# v0.7.0 LLM Epoch Structuring Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace auto-promote placeholders in epoch Phase 2 (L2→L3 consolidation) and Phase 5 (reconsolidation) with LLM-driven structuring, so capture_log entries get refined type/tags and engram content gets updated from recon context.

**Architecture:** The infrastructure is already in place — `LLMClient` (llm.py), 7-phase epoch (epoch.py), cognitive debt (repository.py), rigidity gating (rigidity.py), and `llm_client` threading through `run_epoch_phases()`. This plan adds: (1) prompt templates for Phase 2 structuring and Phase 5 reconsolidation, (2) wiring LLM calls into existing phase functions, (3) graceful fallback to cognitive debt on LLM failure. No schema changes needed.

**Tech Stack:** Python 3.10+, SQLite, existing LLMClient (OpenAI-compatible), pytest

### Key design decisions

1. **Prompts in separate module** — `prompts.py` holds prompt templates as pure functions returning strings. Easy to test, iterate, and review independently.
2. **Batch structuring in Phase 2** — Send all pending capture_log items to LLM in one call (up to a configurable batch size). LLM returns JSON array of structured results. This reduces API calls vs one-per-item.
3. **Rigidity gating in Phase 5** — Only pass engrams with `can_modify_content(rigidity)=True` to LLM. High-rigidity engrams skip reconsolidation entirely.
4. **Fallback on any LLM error** — Catch exceptions, defer to cognitive debt, continue epoch (don't abort). This preserves the existing degradation behavior.
5. **No token counting in v0.7.0** — Listed as optional in roadmap. Can be added later without architectural changes.

```python
# src/memento/prompts.py
"""LLM prompt templates for Epoch structuring and reconsolidation."""

from typing import Optional


VALID_TYPES = ["decision", "insight", "convention", "debugging", "preference", "fact"]


def build_structuring_prompt(items: list[dict]) -> Optional[str]:
    """Build prompt for Phase 2: L2→L3 structuring.

    Input: list of capture_log dicts with id, content, type, tags.
    Output: prompt string asking LLM to return JSON array of structured results.
    Returns None if items is empty.
    """
    if not items:
        return None

    items_text = "\n".join(
        f"- [{item['id']}] {item['content']}"
        for item in items
    )

    return f"""Analyze the following memory captures and return a JSON array.

For each item, determine:
1. "type": the most appropriate category from {VALID_TYPES}
2. "tags": a list of 2-5 short keyword tags (lowercase)
3. "content": the original content, optionally cleaned up for clarity (fix typos, normalize formatting, but preserve meaning)
4. "merge_with": if this item is a near-duplicate of another item in the list, set this to the other item's id; otherwise null

Input items:
{items_text}

Return ONLY a JSON array. Each element must have: "id" (original), "type", "tags", "content", "merge_with".
Example:
[
  {{"id": "c1", "type": "fact", "tags": ["redis", "cache", "ttl"], "content": "Redis needs TTL config for cache entries", "merge_with": null}},
  {{"id": "c2", "type": "preference", "tags": ["ui", "dark-mode"], "content": "User prefers dark mode", "merge_with": null}}
]"""


def build_reconsolidation_prompt(
    engram_content: str,
    engram_type: str,
    recon_contexts: list[str],
) -> Optional[str]:
    """Build prompt for Phase 5: reconsolidation.

    Input: engram content + list of query contexts from recon_buffer.
    Output: prompt asking LLM to refine the engram content based on new context.
    Returns None if recon_contexts is empty.
    """
    if not recon_contexts:
        return None

    contexts_text = "\n".join(f"- {ctx}" for ctx in recon_contexts)

    return f"""You are refining a stored memory based on new context from recent interactions.

Current memory ({engram_type}):
"{engram_content}"

New context from recent queries:
{contexts_text}

If the new context provides meaningful additional information, return an updated version of the memory.
If the memory is already accurate and complete, return it unchanged.

Return ONLY a JSON object with:
- "content": the refined memory text (keep concise, under 200 characters)
- "changed": true if you modified the content, false if unchanged

Do not add information that isn't supported by the context. Do not change the meaning."""
```

```python
def _phase2_l2_consolidation(conn, epoch_id, mode, seal_ts, llm_client=None):
    """Phase 2: L2 → L3 consolidation."""
    
    # 1. Fetch fresh capture_logs
    rows = conn.execute(
        """SELECT * FROM capture_log
           WHERE epoch_id IS NULL AND created_at < ?""",
        (seal_ts,),
    ).fetchall()
    
    # 2. Fetch pending cognitive debt for consolidation
    debt_rows = conn.execute(
        """SELECT c.* FROM capture_log c
           JOIN cognitive_debt d ON json_extract(d.payload, '$.capture_log_id') = c.id
           WHERE d.type='pending_consolidation' AND d.status='open' AND c.epoch_id IS NULL"""
    ).fetchall()

    if not rows and not debt_rows:
        return

    # Deduplicate items in case a row is both fresh and in debt
    items_dict = {row["id"]: dict(row) for row in rows + debt_rows}
    items = list(items_dict.values())

    if mode == "full" and llm_client is not None:
        # LLM-driven structuring
        from memento.prompts import build_structuring_prompt
        
        # Batch items to prevent context window explosion (e.g. 20 per batch)
        BATCH_SIZE = 20
        for i in range(0, len(items), BATCH_SIZE):
            batch = items[i:i+BATCH_SIZE]
            
            prompt = build_structuring_prompt(batch)
            if not prompt: continue

            llm_results = None
            try:
                llm_results = llm_client.generate_json(prompt)
            except Exception:
                pass

            if llm_results and isinstance(llm_results, list):
                # IMPORTANT: Build lookup by ID, never by array index!
                result_map = {res.get("id"): res for res in llm_results if isinstance(res, dict) and res.get("id")}
                
                for item in batch:
                    llm_item = result_map.get(item["id"])
                    
                    if llm_item and llm_item.get("merge_with"):
                        # Safety check: does the merge target actually exist in this batch?
                        if llm_item["merge_with"] in result_map:
                            conn.execute(
                                "UPDATE capture_log SET epoch_id=?, disposition='merged' WHERE id=?",
                                (epoch_id, item["id"]),
                            )
                            resolve_debt(conn, "pending_consolidation", {"capture_log_id": item["id"]})
                            continue
                        else:
                            # Invalid merge target, ignore the merge instruction
                            llm_item["merge_with"] = None

                    if llm_item:
                        import hashlib
                        import json as _json
                        
                        # Create a new dict to avoid mutating the original row
                        processed_item = item.copy()
                        processed_item["type"] = llm_item.get("type", item.get("type", "fact"))
                        
                        # If LLM modified content, we MUST update content_hash and clear embedding
                        new_content = llm_item.get("content")
                        if new_content and new_content != item["content"]:
                            processed_item["content"] = new_content
                            processed_item["content_hash"] = hashlib.sha256(new_content.encode()).hexdigest()
                            processed_item["embedding"] = None
                            processed_item["embedding_dim"] = None
                            processed_item["embedding_pending"] = 1
                        
                        if llm_item.get("tags"):
                            processed_item["tags"] = _json.dumps(llm_item["tags"])

                        plan = TransitionPlan(
                            engram_id=None,
                            capture_log_id=item["id"],
                            from_state="buffered",
                            to_state="consolidated",
                            transition="T1",
                            reason="llm-structured (v0.7.0)",
                            epoch_id=epoch_id,
                        )
                        apply_l2_to_l3(conn, plan, processed_item)
                        resolve_debt(conn, "pending_consolidation", {"capture_log_id": item["id"]})
                    else:
                        # LLM missed this specific item — defer to debt
                        defer_to_debt(conn, "pending_consolidation", {"capture_log_id": item["id"]}, epoch_id)
            else:
                # LLM failed — defer batch to debt
                for item in batch:
                    defer_to_debt(conn, "pending_consolidation", {"capture_log_id": item["id"]}, epoch_id)
    elif mode == "full":
        # No LLM client — auto-promote (v0.5.0 behavior)
        for item in items:
            plan = TransitionPlan(
                engram_id=None,
                capture_log_id=item["id"],
                from_state="buffered",
                to_state="consolidated",
                transition="T1",
                reason="auto-promote (no llm)",
                epoch_id=epoch_id,
            )
            apply_l2_to_l3(conn, plan, item)
            resolve_debt(conn, "pending_consolidation", {"capture_log_id": item["id"]})
    else:
        # Light mode — defer all
        for item in items:
            defer_to_debt(conn, "pending_consolidation", {"capture_log_id": item["id"]}, epoch_id)

    conn.commit()
```
